1 Simple Explanation

An AI Agent is an LLM that can take actions in the world — not just generate text, but actually do things: search the web, run code, read files, send emails, call APIs. The LLM acts as the "brain" that decides what to do; tools are its "hands."

Think of the difference between a person who can only advise you and a person who can actually go do the task. Traditional LLMs advise; agents act.

🤖 Agent = LLM + Tools + Memory + Planning Loop
The agent observes the environment, thinks about what to do, takes an action, observes the result, and repeats until the task is done.

3 Building an Agent from Scratch

python
import json
from openai import OpenAI

client = OpenAI()

# ── Define Tools ──────────────────────────────────────────
def web_search(query: str) -> str:
    # In production, use Serper, Tavily, or Brave Search API
    return f"Search results for '{query}': [Mock] Tesla shares up 8% on strong Q3..."

def calculator(expression: str) -> str:
    try:
        result = eval(expression)  # use safer math library in production
        return str(result)
    except Exception as e:
        return f"Error: {e}"

def get_current_time() -> str:
    from datetime import datetime
    return datetime.now().strftime("%Y-%m-%d %H:%M")

# Tool registry — map name to function
TOOLS = {
    "web_search": web_search,
    "calculator": calculator,
    "get_current_time": get_current_time,
}

# OpenAI tool schema
TOOL_SCHEMAS = [
    {
        "type": "function",
        "function": {
            "name": "web_search",
            "description": "Search the web for current information",
            "parameters": {
                "type": "object",
                "properties": {
                    "query": {"type": "string", "description": "Search query"}
                },
                "required": ["query"]
            }
        }
    },
    {
        "type": "function",
        "function": {
            "name": "calculator",
            "description": "Evaluate mathematical expressions",
            "parameters": {
                "type": "object",
                "properties": {
                    "expression": {"type": "string", "description": "Math expression"}
                },
                "required": ["expression"]
            }
        }
    }
]

# ── Agent Loop ─────────────────────────────────────────────
def run_agent(task: str, max_steps: int = 10) -> str:
    """Run the ReAct agent loop."""
    messages = [
        {"role": "system", "content": "You are a helpful AI agent. Use tools to complete tasks. Think step by step."},
        {"role": "user", "content": task}
    ]

    for step in range(max_steps):
        print(f"
--- Step {step + 1} ---")

        response = client.chat.completions.create(
            model="gpt-4o",
            messages=messages,
            tools=TOOL_SCHEMAS,
            tool_choice="auto"
        )

        msg = response.choices[0].message
        messages.append(msg)

        # No tool call = agent is done
        if not msg.tool_calls:
            print(f"✓ Agent finished: {msg.content}")
            return msg.content

        # Execute each tool call
        for tool_call in msg.tool_calls:
            fn_name = tool_call.function.name
            fn_args = json.loads(tool_call.function.arguments)

            print(f"→ Calling: {fn_name}({fn_args})")
            result = TOOLS[fn_name](**fn_args)
            print(f"← Result: {result[:100]}...")

            messages.append({
                "role": "tool",
                "tool_call_id": tool_call.id,
                "content": result,
            })

    return "Max steps reached"

# Run it!
result = run_agent("Search for Tesla stock news and calculate: 250 * 1.08")
print("
Final Answer:", result)
